--記事を5千件作成
INSERT INTO articles(name, content)
SELECT format('テストユーザー%s', i),format('書き込みテスト%s',i)
FROM generate_series(1,5000) as i;

--コメントを1万件作成(記事idは1~100でランダム)
INSERT INTO comments(name, content, article_id)
SELECT format('テストユーザー%s', (random() * 100)::int + 1),format('コメントテスト%s',i), (random() * 100)::int + 1
FROM generate_series(1,10000) AS i;
